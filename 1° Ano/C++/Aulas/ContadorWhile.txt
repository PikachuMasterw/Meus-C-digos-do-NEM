#include <iostream>
using namespace std;

int main() {
    int x, r, cont = 1;
    
    while(cont <= 5){
        cout<<"Digite um número: ";
        cin>>x;
        
        r = x * 3;
        
        cout<<"R = "<<r;
        cont = cont + 1;
        cout<<"\nQTD de laços: "<<cont;
    }
    
    return 0;
}
