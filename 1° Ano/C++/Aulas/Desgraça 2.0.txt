#include <iostream>
using namespace std;

int main(){
    float novosalario, salario;
    
    cout<<"Digite o salário: $";

    if( salario < 500){

        cout<<"O salário novo foi: $" << novosalario; 
    }

         novosalario = salario * 1.1;
      else( salario > 1000);
         novosalario = salario * 1.05;
            
    cout<<"O salário novo foi: $" << novosalario << endl;
    }
  return 0;
 }
