#include <iostream>
using namespace std;

int main() {
    int num;
    
    cout<<"Digite um número: ";
    cin>>num;
    
    if(num % 5 == 0){
        cout<<"Seu número é multiplo de 5! ("<<num<<")";
    }
    
    else{
        cout<<"Seu número não é multiplo de 5! ("<<num<<")";
    }
    return 0;
}
